//S3 bucket name : s3://snowflake-practice100/test2.csv
//Role ARN : arn:aws:iam::241533163202:role/snowflake-practice100

// snowpipe

create database demo;
create schema snowflake_practice100;

// integration between snowflake and s3
create or replace storage integration aws_s3_integration
type = external_stage,
storage_provider = 'S3',
enabled = TRUE,
storage_aws_role_arn = 'arn:aws:iam::241533163202:role/snowflake-practice100',
storage_allowed_locations = ('s3://snowflake-practice100/test2.csv')
storage_aws_external_id = 'HV94140_SFCRole=3_uYA+Tm5qK3DxCgGafIY0rY1jKfo=';

show integrations;

desc integration aws_s3_integration;

// use this if you are on other role
grant usage on integration aws_s3_integration to role accountadmin;

// create file format to store files on external stage in snowflake
create or replace file format demo_format
type = 'csv'
field_delimiter = '\t' -- Use tab delimiter for your file
skip_header = 1; -- Skip the header row (name, age, experience, salary)

// create stage
create or replace stage demo_aws_stage
storage_integration = aws_s3_integration
file_format = demo_format
URL = 's3://snowflake-practice100/test2.csv';

list @demo_aws_stage;

remove @demo_aws_stage/test2.csv;

create or replace temporary table demo_table(
name string,
age integer,
experience integer,
salary integer
);

select * from demo_table;

// copy aws file content into the table
copy into demo_table
from @demo_aws_stage/test2.csv
file_format = (format_name = demo_format);

select * from demo_table;

-- scenario 1 - read 100 files at once
copy into demo_table
from @demo_aws_stage/
file_format = (format_name = demo_format)
on_error = 'Skip_file'; -- skips the only the file producing error, load others
on_error = 'Continue'; -- skips only bad records on that particular file and loads the rest of the file
on_error = 'abort'; -- stops the entire process if any error is found

-- scenario 2 - Copy executed with 0 files processed -snowflake does not load the same data multiple times
